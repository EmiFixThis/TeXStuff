\begin{defn}[\textbf{Represented As}]
An expression describing a form in which a quantity can be written.
\end{defn}
\medskip
\begin{exmp}
All primes $p > 3$ can be \textit{''represented as''} $6n \pm 1$.
\end{exmp}
