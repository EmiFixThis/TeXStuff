% Notes

% system note : requires 4 arguments, not for user
% 1 = Title,
% 2 = Content,
% 3 = Page height
% 4 = Page color
\newcommand{\systemNote}[4]{
	\pdfpageheight=#3
	\pagecolor{#4}\afterpage{\nopagecolor}
	\rhead{\small \currentCategory}
	\subsection*{#1}
	#2
	\clearpage
}
